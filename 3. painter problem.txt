 long long int p;

   long long int min=0;

    for(long long int j=0;j<p;j++){
        cin>>a[j];

            min=a[j];
        }
        sum=sum+a[j];
    }
long long int l=min;

long long int w;

 mid=(l+r)/2;
 w=0;

w=w+a[i];

}

}
